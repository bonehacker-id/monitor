  <script>

    const REFRESH_INTERVAL = 90000;

    const STATUS = { WAITING: 'waiting', IN_TREATMENT: 'in-treatment', COMPLETED: 'completed' };

    const BRANCH_CONFIG = {

      name: "Purwokerto",

      url: "https://script.google.com/macros/s/AKfycbzKpxIOF2VABpGLbDRj9-9Bi2Gwi-urycCnScYhbn26Fy2GQqQGueMyqR6X7F8fkQaO/exec"

    };


    function init() {

        updateClockAndDate();

        setInterval(updateClockAndDate, 1000);

        fetchData();

        setInterval(fetchData, REFRESH_INTERVAL);

    }


    function updateClockAndDate() {

        const now = new Date();

        const optionsDate = { weekday: 'long', day: 'numeric', month: 'long', year: 'numeric' };

        document.getElementById('date').textContent = now.toLocaleDateString('id-ID', optionsDate);

        document.getElementById('clock').textContent = now.toLocaleTimeString('id-ID', { hour12: false });

    }


    async function fetchData() {

        try {

            const response = await fetch(BRANCH_CONFIG.url);

            const data = await response.json();

            updateUI(data);

        } catch (error) {

            console.error("Fetch Error:", error);

        }

        document.getElementById('lastUpdated').textContent = new Date().toLocaleTimeString("id-ID") + " WIB";

    }


    function updateUI(patients) {

        const safePatients = patients || [];

        

        const waitingList = safePatients.filter(p => p.status === STATUS.WAITING);

        const treatingList = safePatients.filter(p => p.status === STATUS.IN_TREATMENT);

        const completedList = safePatients.filter(p => p.status === STATUS.COMPLETED);


        // Update Summary (Horizontal border-top dihapus di sini)

        document.getElementById('summaryContainer').innerHTML = `

            <div class="grid grid-cols-3 gap-8 text-center">

                <div class="pt-2">

                    <span class="text-5xl font-bold status-waiting">${waitingList.length}</span>

                    <p class="text-sm tracking-widest font-bold opacity-70 uppercase">Menunggu</p>

                </div>

                <div class="pt-2">

                    <span class="text-5xl font-bold status-in-treatment">${treatingList.length}</span>

                    <p class="text-sm tracking-widest font-bold opacity-70 uppercase">Terapi</p>

                </div>

                <div class="pt-2">

                    <span class="text-5xl font-bold status-completed">${completedList.length}</span>

                    <p class="text-sm tracking-widest font-bold opacity-70 uppercase">Selesai</p>

                </div>

            </div>

        `;


        renderWaitingList(waitingList);

        renderStandardList('list-treating', treatingList);


        // Hanya 4 terakhir untuk kolom selesai

        const last4Completed = [...completedList].reverse().slice(0, 4);

        renderStandardList('list-completed', last4Completed);

    }


    function renderWaitingList(list) {

        const container = document.getElementById('list-waiting');

        if (list.length === 0) {

            container.innerHTML = `<div class="p-8 text-center opacity-50 italic">Kosong</div>`;

            container.style.animation = 'none';

            return;

        }


        const itemsToRender = [...list, ...list];

        const originalLength = list.length;


        container.innerHTML = itemsToRender.map((p, index) => {

            const isJunction = index === originalLength - 1;

            return `

                <div class="patient-row ${isJunction ? 'border-junction' : ''}">

                    <div class="flex items-center gap-4">

                        <span class="text-2xl font-black bg-gray-100 dark:bg-zinc-800 px-3 py-1 rounded-lg">${p.waitingNumber}</span>

                        <span class="text-2xl font-semibold">${p.patientName}</span>

                    </div>

                    <span class="text-lg font-medium text-orange-500">Antrian: ${p.queuePosition}</span>

                </div>

            `;

        }).join('');


        const duration = Math.max(12, originalLength * 3);

        container.style.animation = originalLength > 6 ? `scroll-up ${duration}s linear infinite` : 'none';

    }


    function renderStandardList(elementId, list) {

        const container = document.getElementById(elementId);

        if (list.length === 0) {

            container.innerHTML = `<div class="p-8 text-center opacity-50 italic">Kosong</div>`;

            return;

        }


        container.innerHTML = list.map(p => `

            <div class="patient-row">

                <div class="flex items-center gap-4">

                    <span class="text-2xl font-black bg-gray-100 dark:bg-zinc-800 px-3 py-1 rounded-lg">${p.waitingNumber}</span>

                    <span class="text-2xl font-semibold">${p.patientName}</span>

                </div>

            </div>

        `).join('');

    }


    document.addEventListener('DOMContentLoaded', init);
